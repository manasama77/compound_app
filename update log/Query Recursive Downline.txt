with recursive tree (id, fullname, id_upline) as (
  select     id,
             fullname,
             id_upline
  from       et_member
  where      id_upline = '01df9753-cf18-11eb-9d7c-2f0589a292f2'
  union all
  select     p.id,
             p.fullname,
             p.id_upline
  from       et_member p
  inner join tree
          on p.id_upline = tree.id
)
select * from tree;
